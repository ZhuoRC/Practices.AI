# 阶段1：脚本处理系统分析和实现计划

## 1. 需求分析

### 核心目标
- 将长篇文章转换为适合视频的旁白脚本
- 实现智能分段，考虑口语表达和视频节奏
- 提取关键词用于视频高亮显示
- 生成结构化的脚本数据，支持后续处理

### 输入输出
**输入：**
- 原始文本文件 (`scripts.txt`)
- 用户配置参数（分段长度、语言风格、目标时长等）

**输出：**
- 结构化脚本JSON文件
- 分段信息
- 关键词列表
- 时间估算

```python
# script_processor/ai_service.py
import aiohttp
import json
from typing import Dict, Any

class LocalAIService:
    """本地AI服务客户端"""
    
    def __init__(self, base_url: str = "http://localhost:8060"):
        self.base_url = base_url
        self.session = None
    
    async def __aenter__(self):
        self.session = aiohttp.ClientSession()
        return self
    
    async def __aexit__(self, exc_type, exc_val, exc_tb):
        if self.session:
            await self.session.close()
    
    async def rewrite_text_sync(self, text: str, style: str = "conversational") -> str:
        """同步改写文本"""
        url = f"{self.base_url}/api/rewriter/rewrite"
        
        payload = {
            "text": text,
            "style": style,
            "target_audience": "general",
            "tone": "natural"
        }
        
        async with self.session.post(url, json=payload) as response:
            if response.status == 200:
                data = await response.json()
                return data.get("rewritten_text", text)
            else:
                error_data = await response.json()
                raise AIServiceError(f"API调用失败: {error_data}")
    
    async def rewrite_text_async(self, text: str, style: str = "conversational") -> str:
        """异步改写文本"""
        url = f"{self.base_url}/api/rewriter/rewrite/async"
        
        payload = {
            "text": text,
            "style": style,
            "target_audience": "general",
            "tone": "natural"
        }
        
        # 启动异步任务
        async with self.session.post(url, json=payload) as response:
            if response.status == 202:
                data = await response.json()
                task_id = data.get("task_id")
                
                # 轮询任务状态
                return await self._poll_task_completion(task_id)
            else:
                error_data = await response.json()
                raise AIServiceError(f"异步任务创建失败: {error_data}")
    
    async def _poll_task_completion(self, task_id: str) -> str:
        """轮询任务完成状态"""
        status_url = f"{self.base_url}/api/rewriter/status/{task_id}"
        
        while True:
            async with self.session.get(status_url) as response:
                if response.status == 200:
                    data = await response.json()
                    status = data.get("status")
                    
                    if status == "completed":
                        return data.get("result", {}).get("rewritten_text", "")
                    elif status == "failed":
                        error_msg = data.get("error", "未知错误")
                        raise AIServiceError(f"文本改写失败: {error_msg}")
                    elif status in ["pending", "processing"]:
                        await asyncio.sleep(2)  # 等待2秒后重试
                        continue
                    else:
                        raise AIServiceError(f"未知任务状态: {status}")
                else:
                    raise AIServiceError(f"状态查询失败: {response.status}")
    
    async def get_available_styles(self) -> Dict[str, Any]:
        """获取可用的改写风格"""
        url = f"{self.base_url}/api/rewriter/styles"
        
        async with self.session.get(url) as response:
            if response.status == 200:
                return await response.json()
            else:
                raise AIServiceError(f"获取风格失败: {response.status}")

class AIServiceError(Exception):
    """AI服务异常"""
    pass
```

```python
# script_processor/segmenter.py
import re
from typing import List, Tuple

class TextSegmenter:
    def __init__(self, target_duration: float = 45.0):
        self.target_duration = target_duration
    
    def segment_text(self, text: str) -> List[Segment]:
        """主要分段方法"""
        # 1. 预处理文本
        cleaned_text = self._preprocess_text(text)
        
        # 2. 找到候选分段点
        break_points = self._find_break_points(cleaned_text)
        
        # 3. 优化分段点
        optimized_segments = self._optimize_segments(cleaned_text, break_points)
        
        return optimized_segments
    
    def _find_break_points(self, text: str) -> List[int]:
        """找到所有可能的分段点"""
        break_points = []
        
        # 句号、问号、感叹号
        sentence_endings = re.finditer(r'[。！？]', text)
        for match in sentence_endings:
            break_points.append(match.end())
        
        # 段落分隔
        paragraph_breaks = re.finditer(r'\n\s*\n', text)
        for match in paragraph_breaks:
            break_points.append(match.end())
        
        return sorted(list(set(break_points)))
    
    def _estimate_segment_duration(self, text: str) -> float:
        """估算段落时长"""
        char_count = len(text.replace(" ", ""))
        return char_count / 3.0  # 3字/秒
```

### 步骤4：关键词提取实现
```python
# script_processor/keyword_extractor.py
import jieba
import jieba.analyse
from collections import Counter

class ChineseKeywordExtractor:
    def __init__(self):
        jieba.initialize()
        
    def extract_keywords(self, text: str, top_k: int = 10) -> List[Keyword]:
        """提取中文关键词"""
        # 使用TF-IDF提取关键词
        tfidf_keywords = jieba.analyse.extract_tags(text, topK=top_k, withWeight=True)
        
        # 使用TextRank提取关键词
        textrank_keywords = jieba.analyse.textrank(text, topK=top_k, withWeight=True)
        
        # 合并和排序结果
        combined_keywords = self._combine_keywords(tfidf_keywords, textrank_keywords)
        
        return combined_keywords
    
    def _combine_keywords(self, tfidf_list: List[Tuple], textrank_list: List[Tuple]) -> List[Keyword]:
        """合并两种算法的结果"""
        keyword_scores = {}
        
        for word, score in tfidf_list:
            keyword_scores[word] = score * 0.6  # TF-IDF权重0.6
        
        for word, score in textrank_list:
            if word in keyword_scores:
                keyword_scores[word] += score * 0.4  # TextRank权重0.4
            else:
                keyword_scores[word] = score * 0.4
        
        # 排序并转换为Keyword对象
        sorted_keywords = sorted(keyword_scores.items(), key=lambda x: x[1], reverse=True)
        
        return [
            Keyword(word=word, importance=score, positions=[])
            for word, score in sorted_keywords
        ]
```
