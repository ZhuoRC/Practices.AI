**输入：**
- 音频段文件列表（来自阶段2）
- 视频段文件列表（来自阶段3）

**输出：**
- 完整的合并视频文件
- 音频同步日志
- 质量检查报告
- 元数据文件

### 2.1 FFmpeg处理器

#### 核心FFmpeg集成
```python
# aggregator/ffmpeg_processor.py
import subprocess
import json
import re
from pathlib import Path
from typing import List, Dict, Optional, Tuple
import asyncio

class FFmpegProcessor:
    """FFmpeg处理器核心类"""
    
    def __init__(self, ffmpeg_path: str = "ffmpeg", ffprobe_path: str = "ffprobe"):
        self.ffmpeg_path = ffmpeg_path
        self.ffprobe_path = ffprobe_path
        self.validate_ffmpeg_installation()
    
    def validate_ffmpeg_installation(self):
        """验证FFmpeg安装"""
        try:
            result = subprocess.run(
                [self.ffmpeg_path, "-version"],
                capture_output=True,
                text=True,
                timeout=10
            )
            if result.returncode != 0:
                raise FFmpegError("FFmpeg未正确安装")
        except FileNotFoundError:
            raise FFmpegError("找不到FFmpeg可执行文件")
    
    async def get_video_info(self, file_path: str) -> Dict[str, Any]:
        """获取视频文件信息"""
        cmd = [
            self.ffprobe_path,
            "-v", "quiet",
            "-print_format", "json",
            "-show_format",
            "-show_streams",
            file_path
        ]
        
        process = await asyncio.create_subprocess_exec(
            *cmd,
            stdout=asyncio.subprocess.PIPE,
            stderr=asyncio.subprocess.PIPE
        )
        
        stdout, stderr = await process.communicate()
        
        if process.returncode != 0:
            raise FFmpegError(f"获取视频信息失败: {stderr.decode()}")
        
        return json.loads(stdout.decode())
    
    async def merge_segments_with_audio(self, 
                                      video_segments: List[VideoSegment],
                                      audio_segments: List[AudioSegment],
                                      output_path: str,
                                      config: MergeConfig) -> MergeResult:
        """合并视频和音频段落"""
        logger.info(f"开始合并 {len(video_segments)} 个视频段落")
        
        # 1. 创建文件列表
        video_list_file = await self._create_file_list(video_segments, "video")
        audio_list_file = await self._create_file_list(audio_segments, "audio")
        
        # 2. 生成FFmpeg命令
        ffmpeg_cmd = await self._build_merge_command(
            video_list_file, audio_list_file, output_path, config
        )
        
        # 3. 执行合并
        result = await self._execute_ffmpeg_command(ffmpeg_cmd)
        
        # 4. 清理临时文件
        await self._cleanup_temp_files([video_list_file, audio_list_file])
        
        # 5. 验证输出
        await self._validate_output(output_path)
        
        return MergeResult(
            output_path=output_path,
            duration=await self._get_total_duration(output_path),
            file_size=Path(output_path).stat().st_size,
            success=result.returncode == 0,
            logs=result.stderr.decode() if result.stderr else ""
        )
    
    async def _create_file_list(self, segments: List, segment_type: str) -> str:
        """创建FFmpeg文件列表"""
        list_path = f"./temp/{segment_type}_list.txt"
        
        with open(list_path, 'w', encoding='utf-8') as f:
            for segment in sorted(segments, key=lambda x: getattr(x, 'index', x.segment_index)):
                if segment_type == "video":
                    file_path = segment.file_path
                    duration = segment.duration
                else:  # audio
                    file_path = segment.file_path
                    duration = segment.duration
                
                # 格式: file 'path' duration X
                f.write(f"file '{file_path}'\n")
                if duration:
                    f.write(f"duration {duration}\n")
        
        return list_path
    
    async def _build_merge_command(self, 
                                 video_list: str, 
                                 audio_list: str, 
                                 output_path: str, 
                                 config: MergeConfig) -> List[str]:
        """构建FFmpeg合并命令"""
        cmd = [
            self.ffmpeg_path,
            "-f", "concat",
            "-safe", "0",
            "-i", video_list,
            "-f", "concat",
            "-safe", "0",
            "-i", audio_list,
            "-c:v", config.video_codec,
            "-c:a", config.audio_codec,
            "-preset", config.preset,
            "-crf", str(config.crf),
            "-r", str(config.fps),
            "-pix_fmt", config.pixel_format
        ]
        
        # 添加转场效果
        if config.enable_transitions:
            transition_filter = await self._build_transition_filter(config.transition_type)
            cmd.extend(["-vf", transition_filter])
        
        # 添加音频同步
        if config.audio_sync_enabled:
            cmd.extend(["-async", "1", "-vsync", "1"])
        
        # 添加输出选项
        cmd.extend([
            "-map", "0:v",  # 使用第一个输入的视频流
            "-map", "1:a",  # 使用第二个输入的音频流
            "-shortest",    # 以较短的流为准
            "-y",           # 覆盖输出文件
            output_path
        ])
        
        return cmd
    
    async def _build_transition_filter(self, transition_type: str) -> str:
        """构建转场滤镜"""
        filters = []
        
        if transition_type == "fade":
            filters.extend([
                "fade=type=in:duration=0.5:alpha=1",
                "fade=type=out:duration=0.5:alpha=1:start_frame=-1"
            ])
        elif transition_type == "crossfade":
            filters.append("xfade=transition=fade:duration=0.5:offset=0")
        elif transition_type == "dissolve":
            filters.append("blend=all_expr='A*(if(gte(T,0.5),1,T/0.5))+B*(1-(if(gte(T,0.5),1,T/0.5)))'")
        
        return ",".join(filters)
    
    async def _execute_ffmpeg_command(self, cmd: List[str]) -> subprocess.CompletedProcess:
        """执行FFmpeg命令"""
        logger.info(f"执行FFmpeg命令: {' '.join(cmd)}")
        
        process = await asyncio.create_subprocess_exec(
            *cmd,
            stdout=asyncio.subprocess.PIPE,
            stderr=asyncio.subprocess.PIPE
        )
        
        stdout, stderr = await process.communicate()
        
        if process.returncode != 0:
            error_msg = stderr.decode() if stderr else "未知错误"
            logger.error(f"FFmpeg执行失败: {error_msg}")
            raise FFmpegError(f"FFmpeg执行失败: {error_msg}")
        
        return process
    
    async def _validate_output(self, output_path: str):
        """验证输出文件"""
        if not Path(output_path).exists():
            raise FFmpegError("输出文件不存在")
        
        # 获取文件信息
        info = await self.get_video_info(output_path)
        
        # 检查是否有视频和音频流
        video_streams = [s for s in info['streams'] if s['codec_type'] == 'video']
        audio_streams = [s for s in info['streams'] if s['codec_type'] == 'audio']
        
        if not video_streams:
            raise FFmpegError("输出文件没有视频流")
        
        if not audio_streams:
            raise FFmpegError("输出文件没有音频流")
        
        logger.info(f"输出验证成功: {output_path}")
```

### 2.2 音视频同步器

#### 精确同步算法
```python
# aggregator/synchronizer.py
import numpy as np
from typing import List, Dict, Tuple, Optional
from dataclasses import dataclass

@dataclass
class SyncPoint:
    video_time: float
    audio_time: float
    segment_id: str
    confidence: float

class AudioVideoSynchronizer:
    """音视频同步器"""
    
    def __init__(self, tolerance_ms: float = 100.0):
        self.tolerance_ms = tolerance_ms
        self.sync_points: List[SyncPoint] = []
    
    async def synchronize_segments(self, 
                                 video_segments: List[VideoSegment],
                                 audio_segments: List[AudioSegment]) -> SyncResult:
        """同步视频和音频段落"""
        logger.info("开始音视频同步")
        
        # 1. 创建时间轴映射
        video_timeline = self._create_timeline(video_segments)
        audio_timeline = self._create_timeline(audio_segments)
        
        # 2. 检测同步点
        sync_points = await self._detect_sync_points(video_timeline, audio_timeline)
        
        # 3. 计算偏移量
        offsets = self._calculate_offsets(sync_points)
        
        # 4. 应用同步调整
        adjusted_segments = await self._apply_sync_adjustments(
            video_segments, audio_segments, offsets
        )
        
        # 5. 生成同步报告
        sync_report = self._generate_sync_report(sync_points, offsets)
        
        return SyncResult(
            adjusted_video_segments=adjusted_segments["video"],
            adjusted_audio_segments=adjusted_segments["audio"],
            sync_points=sync_points,
            offsets=offsets,
            report=sync_report
        )
    
    def _create_timeline(self, segments: List) -> Dict[str, Dict[str, float]]:
        """创建时间轴"""
        timeline = {}
        current_time = 0.0
        
        for segment in segments:
            timeline[segment.segment_id] = {
                "start_time": current_time,
                "end_time": current_time + segment.duration,
                "duration": segment.duration
            }
            current_time += segment.duration
        
        return timeline
    
    async def _detect_sync_points(self, 
                                 video_timeline: Dict, 
                                 audio_timeline: Dict) -> List[SyncPoint]:
        """检测同步点"""
        sync_points = []
        
        # 基于段落ID匹配
        for segment_id in video_timeline:
            if segment_id in audio_timeline:
                video_info = video_timeline[segment_id]
                audio_info = audio_timeline[segment_id]
                
                # 计算时间差异
                time_diff = abs(video_info["start_time"] - audio_info["start_time"])
                
                if time_diff <= self.tolerance_ms / 1000.0:
                    sync_point = SyncPoint(
                        video_time=video_info["start_time"],
                        audio_time=audio_info["start_time"],
                        segment_id=segment_id,
                        confidence=1.0 - (time_diff / (self.tolerance_ms / 1000.0))
                    )
                    sync_points.append(sync_point)
        
        return sorted(sync_points, key=lambda x: x.video_time)
    
    def _calculate_offsets(self, sync_points: List[SyncPoint]) -> Dict[str, float]:
        """计算偏移量"""
        if not sync_points:
            return {}
        
        offsets = {}
        
        # 计算每个同步点的偏移
        for sync_point in sync_points:
            offset = sync_point.audio_time - sync_point.video_time
            offsets[sync_point.segment_id] = offset
        
        # 平滑偏移量
        smoothed_offsets = self._smooth_offsets(offsets)
        
        return smoothed_offsets
    
    def _smooth_offsets(self, offsets: Dict[str, float]) -> Dict[str, float]:
        """平滑偏移量"""
        if len(offsets) <= 2:
            return offsets
        
        sorted_segments = sorted(offsets.keys())
        smoothed = {}
        
        for i, segment_id in enumerate(sorted_segments):
            if i == 0 or i == len(sorted_segments) - 1:
                # 保持首尾不变
                smoothed[segment_id] = offsets[segment_id]
            else:
                # 使用相邻点的平均值
                prev_offset = offsets[sorted_segments[i-1]]
                curr_offset = offsets[segment_id]
                next_offset = offsets[sorted_segments[i+1]]
                
                smoothed_offset = (prev_offset + 2*curr_offset + next_offset) / 4
                smoothed[segment_id] = smoothed_offset
        
        return smoothed
    
    async def _apply_sync_adjustments(self, 
                                    video_segments: List[VideoSegment],
                                    audio_segments: List[AudioSegment],
                                    offsets: Dict[str, float]) -> Dict[str, List]:
        """应用同步调整"""
        adjusted_video = []
        adjusted_audio = []
        
        # 创建段落映射
        video_map = {seg.segment_id: seg for seg in video_segments}
        audio_map = {seg.segment_id: seg for seg in audio_segments}
        
        # 调整视频段落
        for segment in video_segments:
            adjusted_segment = VideoSegment(
                segment_id=segment.segment_id,
                file_path=segment.file_path,
                duration=segment.duration,
                resolution=segment.resolution,
                fps=segment.fps,
                file_size=segment.file_size
            )
            adjusted_video.append(adjusted_segment)
        
        # 调整音频段落
        for segment in audio_segments:
            offset = offsets.get(segment.segment_id, 0.0)
            
            adjusted_segment = AudioSegment(
                segment_id=segment.segment_id,
                file_path=segment.file_path,
                duration=segment.duration,
                estimated_duration=segment.estimated_duration,
                voice_used=segment.voice_used,
                provider_used=segment.provider_used,
                file_size=segment.file_size,
                time_offset=offset
            )
            adjusted_audio.append(adjusted_segment)
        
        return {
            "video": adjusted_video,
            "audio": adjusted_audio
        }
    
    def _generate_sync_report(self, sync_points: List[SyncPoint], offsets: Dict[str, float]) -> Dict[str, Any]:
        """生成同步报告"""
        if not sync_points:
            return {
                "total_sync_points": 0,
                "average_offset": 0.0,
                "max_offset": 0.0,
                "sync_quality": "unknown"
            }
        
        offsets_list = list(offsets.values())
        
        return {
            "total_sync_points": len(sync_points),
            "average_offset": np.mean(offsets_list),
            "max_offset": max(abs(o) for o in offsets_list),
            "min_offset": min(offsets_list),
            "sync_quality": self._evaluate_sync_quality(offsets_list),
            "confident_sync_points": len([sp for sp in sync_points if sp.confidence > 0.8])
        }
    
    def _evaluate_sync_quality(self, offsets: List[float]) -> str:
        """评估同步质量"""
        if not offsets:
            return "unknown"
        
        avg_offset = np.mean(np.abs(offsets))
        
        if avg_offset < 0.05:  # 50ms
            return "excellent"
        elif avg_offset < 0.1:   # 100ms
            return "good"
        elif avg_offset < 0.2:   # 200ms
            return "fair"
        else:
            return "poor"
```

# aggregator/transition_processor.py
from enum import Enum
from typing import List, Dict, Any
import numpy as np

class TransitionType(Enum):
    FADE = "fade"
    CROSSFADE = "crossfade"
    DISSOLVE = "dissolve"
    SLIDE = "slide"
    ZOOM = "zoom"
    WIPE = "wipe"

class TransitionProcessor:
    """转场效果处理器"""
    
    def __init__(self):
        self.transition_presets = self._load_transition_presets()
    
    async def apply_transitions(self, 
                              segments: List[VideoSegment],
                              transition_config: TransitionConfig) -> List[ProcessedSegment]:
        """应用转场效果"""
        processed_segments = []
        
        for i, segment in enumerate(segments):
            # 确定转场类型
            transition_type = self._determine_transition_type(
                i, len(segments), transition_config
            )
            
            # 生成转场滤镜
            transition_filter = await self._generate_transition_filter(
                transition_type, transition_config
            )
            
            # 创建处理后的段落
            processed_segment = ProcessedSegment(
                original_segment=segment,
                transition_type=transition_type,
                transition_filter=transition_filter,
                transition_duration=transition_config.duration
            )
            
            processed_segments.append(processed_segment)
        
        return processed_segments
    
    def _determine_transition_type(self, 
                                  index: int, 
                                  total_segments: int,
                                  config: TransitionConfig) -> TransitionType:
        """确定转场类型"""
        # 最后一个段落不需要输出转场
        if index == total_segments - 1:
            return TransitionType.FADE  # 只有淡入
        
        # 根据配置决定
        if config.auto_select:
            # 基于段落内容自动选择
            return self._auto_select_transition(index, total_segments)
        else:
            return config.default_type
    
    def _auto_select_transition(self, index: int, total_segments: int) -> TransitionType:
        """自动选择转场类型"""
        # 简单的轮换策略
        transitions = [
            TransitionType.FADE,
            TransitionType.CROSSFADE,
            TransitionType.DISSOLVE
        ]
        return transitions[index % len(transitions)]
    
    async def _generate_transition_filter(self, 
                                        transition_type: TransitionType,
                                        config: TransitionConfig) -> str:
        """生成转场滤镜"""
        if transition_type == TransitionType.FADE:
            return f"fade=type=in:duration={config.duration}:alpha=1"
        
        elif transition_type == TransitionType.CROSSFADE:
            return f"xfade=transition=fade:duration={config.duration}:offset=0"
        
        elif transition_type == TransitionType.DISSOLVE:
            return f"blend=all_expr='A*(if(gte(T,{config.duration/2}),1,T/({config.duration/2})))+B*(1-(if(gte(T,{config.duration/2}),1,T/({config.duration/2}))))'"
        
        elif transition_type == TransitionType.SLIDE:
            return f"slide=duration={config.duration}:mode=left"
        
        elif transition_type == TransitionType.ZOOM:
            return f"zoompan=duration={config.duration}:x='iw/2-(iw/zoom/2)':y='ih/2-(ih/zoom/2)':z='if(lte(on,{config.duration}),zoom+1,1)'"
        
        elif transition_type == TransitionType.WIPE:
            return f"geq=lum='p(X,Y)*(1+0.5*sin(2*PI*X/OW*{config.duration}))'"
        
        else:
            return f"fade=type=in:duration={config.duration}:alpha=1"
    
    def _load_transition_presets(self) -> Dict[str, Dict[str, Any]]:
        """加载转场预设"""
        return {
            "fade": {
                "duration": 0.5,
                "direction": "in",
                "alpha": 1
            },
            "crossfade": {
                "duration": 0.5,
                "offset": 0,
                "transition": "fade"
            },
            "dissolve": {
                "duration": 0.5,
                "blend_mode": "normal"
            },
            "slide": {
                "duration": 0.5,
                "mode": "left"
            },
            "zoom": {
                "duration": 0.5,
                "zoom_factor": 1.5
            },
            "wipe": {
                "duration": 0.5,
                "direction": "horizontal"
            }
        }

# aggregator/quality_controller.py
import cv2
import numpy as np
from pathlib import Path
from typing import Dict, List, Optional

class QualityController:
    """质量控制器"""
    
    def __init__(self, config: QualityConfig):
        self.config = config
        self.quality_thresholds = config.thresholds
    
    async def validate_final_output(self, output_path: str) -> QualityReport:
        """验证最终输出质量"""
        logger.info(f"开始验证输出质量: {output_path}")
        
        # 1. 基本文件检查
        file_check = await self._check_file_integrity(output_path)
        
        # 2. 视频质量分析
        video_quality = await self._analyze_video_quality(output_path)
        
        # 3. 音频质量分析
        audio_quality = await self._analyze_audio_quality(output_path)
        
        # 4. 同步质量检查
        sync_quality = await self._check_audio_video_sync(output_path)
        
        # 5. 综合质量评估
        overall_score = self._calculate_overall_quality(
            file_check, video_quality, audio_quality, sync_quality
        )
        
        # 6. 生成质量报告
        quality_report = QualityReport(
            file_path=output_path,
            file_integrity=file_check,
            video_quality=video_quality,
            audio_quality=audio_quality,
            sync_quality=sync_quality,
            overall_score=overall_score,
            issues=self._detect_quality_issues(
                file_check, video_quality, audio_quality, sync_quality
            ),
            recommendations=self._generate_recommendations(overall_score)
        )
        
        return quality_report
    
    async def _check_file_integrity(self, file_path: str) -> Dict[str, Any]:
        """检查文件完整性"""
        file_path = Path(file_path)
        
        if not file_path.exists():
            return {"valid": False, "error": "文件不存在"}
        
        file_size = file_path.stat().st_size
        
        # 检查文件大小是否合理
        if file_size < 1024:  # 小于1KB
            return {"valid": False, "error": "文件过小"}
        
        # 尝试读取文件头
        try:
            with open(file_path, 'rb') as f:
                header = f.read(12)
                # 检查是否为有效的视频文件头
                if not (header.startswith(b'\x00\x00\x00') or b'ftyp' in header):
                    return {"valid": False, "error": "无效的文件头"}
        except Exception as e:
            return {"valid": False, "error": f"读取文件失败: {e}"}
        
        return {
            "valid": True,
            "file_size": file_size,
            "file_size_mb": round(file_size / (1024 * 1024), 2)
        }
    
    async def _analyze_video_quality(self, file_path: str) -> Dict[str, Any]:
        """分析视频质量"""
        cap = cv2.VideoCapture(file_path)
        
        if not cap.isOpened():
            return {"valid": False, "error": "无法打开视频文件"}
        
        try:
            # 获取基本信息
            fps = cap.get(cv2.CAP_PROP_FPS)
            width = int(cap.get(cv2.CAP_PROP_FRAME_WIDTH))
            height = int(cap.get(cv2.CAP_PROP_FRAME_HEIGHT))
            frame_count = int(cap.get(cv2.CAP_PROP_FRAME_COUNT))
            duration = frame_count / fps if fps > 0 else 0
            
            # 分析视频质量指标
            quality_metrics = await self._analyze_video_frames(cap, sample_frames=100)
            
            return {
                "valid": True,
                "resolution": (width, height),
                "fps": fps,
                "duration": duration,
                "frame_count": frame_count,
                "estimated_bitrate": self._estimate_bitrate(file_path, duration),
                "quality_metrics": quality_metrics
            }
            
        finally:
            cap.release()
    
    async def _analyze_video_frames(self, cap, sample_frames: int = 100) -> Dict[str, float]:
        """分析视频帧质量"""
        total_frames = int(cap.get(cv2.CAP_PROP_FRAME_COUNT))
        step = max(1, total_frames // sample_frames)
        
        brightness_values = []
        contrast_values = []
        blur_values = []
        
        frame_index = 0
        analyzed_frames = 0
        
        while frame_index < total_frames and analyzed_frames < sample_frames:
            cap.set(cv2.CAP_PROP_POS_FRAMES, frame_index)
            ret, frame = cap.read()
            
            if ret:
                # 转换为灰度图
                gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
                
                # 计算亮度
                brightness = np.mean(gray)
                brightness_values.append(brightness)
                
                # 计算对比度
                contrast = np.std(gray)
                contrast_values.append(contrast)
                
                # 计算模糊度
                blur = cv2.Laplacian(gray, cv2.CV_64F).var()
                blur_values.append(blur)
                
                analyzed_frames += 1
            
            frame_index += step
        
        return {
            "average_brightness": np.mean(brightness_values),
            "brightness_variance": np.var(brightness_values),
            "average_contrast": np.mean(contrast_values),
            "average_sharpness": np.mean(blur_values),
            "frames_analyzed": analyzed_frames
        }
    
    async def _analyze_audio_quality(self, file_path: str) -> Dict[str, Any]:
        """分析音频质量"""
        try:
            # 使用FFmpeg分析音频
            cmd = [
                "ffprobe",
                "-v", "quiet",
                "-select_streams", "a:0",
                "-show_entries", "stream=codec_name,sample_rate,channels,duration",
                "-of", "json",
                file_path
            ]
            
            process = await asyncio.create_subprocess_exec(
                *cmd,
                stdout=asyncio.subprocess.PIPE,
                stderr=asyncio.subprocess.PIPE
            )
            
            stdout, stderr = await process.communicate()
            
            if process.returncode != 0:
                return {"valid": False, "error": "音频分析失败"}
            
            data = json.loads(stdout.decode())
            
            if not data.get("streams"):
                return {"valid": False, "error": "没有音频流"}
            
            audio_stream = data["streams"][0]
            
            return {
                "valid": True,
                "codec": audio_stream.get("codec_name"),
                "sample_rate": int(audio_stream.get("sample_rate", 0)),
                "channels": int(audio_stream.get("channels", 0)),
                "duration": float(audio_stream.get("duration", 0))
            }
            
        except Exception as e:
            return {"valid": False, "error": f"音频分析异常: {e}"}
    
    async def _check_audio_video_sync(self, file_path: str) -> Dict[str, Any]:
        """检查音视频同步"""
        try:
            # 获取视频和音频流的时长
            probe_cmd = [
                "ffprobe",
                "-v", "quiet",
                "-show_entries",
                "stream=codec_type,duration",
                "-of", "json",
                file_path
            ]
            
            process = await asyncio.create_subprocess_exec(
                *probe_cmd,
                stdout=asyncio.subprocess.PIPE,
                stderr=asyncio.subprocess.PIPE
            )
            
            stdout, stderr = await process.communicate()
            data = json.loads(stdout.decode())
            
            video_duration = None
            audio_duration = None
            
            for stream in data.get("streams", []):
                if stream["codec_type"] == "video":
                    video_duration = float(stream["duration"])
                elif stream["codec_type"] == "audio":
                    audio_duration = float(stream["duration"])
            
            if video_duration is None or audio_duration is None:
                return {"valid": False, "error": "无法获取流时长"}
            
            duration_diff = abs(video_duration - audio_duration)
            sync_tolerance = 0.1  # 100ms容差
            
            return {
                "valid": duration_diff <= sync_tolerance,
                "video_duration": video_duration,
                "audio_duration": audio_duration,
                "duration_difference": duration_diff,
                "sync_quality": "good" if duration_diff <= sync_tolerance else "poor"
            }
            
        except Exception as e:
            return {"valid": False, "error": f"同步检查失败: {e}"}
    
    def _calculate_overall_quality(self, *quality_checks) -> float:
        """计算总体质量分数"""
        scores = []
        
        for check in quality_checks:
            if isinstance(check, dict) and check.get("valid"):
                scores.append(1.0)
            elif isinstance(check, dict) and not check.get("valid"):
                scores.append(0.0)
        
        return np.mean(scores) if scores else 0.0
    
    def _detect_quality_issues(self, *quality_checks) -> List[str]:
        """检测质量问题"""
        issues = []
        
        for check in quality_checks:
            if isinstance(check, dict):
                if not check.get("valid"):
                    issues.append(check.get("error", "未知质量问题"))
        
        return issues
    
    def _generate_recommendations(self, overall_score: float) -> List[str]:
        """生成改进建议"""
        recommendations = []
        
        if overall_score < 0.8:
            recommendations.append("建议重新生成视频以提高质量")
        
        if overall_score < 0.6:
            recommendations.append("检查输入文件质量")
            recommendations.append("调整编码参数")
        
        if overall_score >= 0.8:
            recommendations.append("视频质量良好")
        
        return recommendations
```

# aggregator/merge_pipeline.py
class VideoMergePipeline:
    """视频合并主管道"""
    
    def __init__(self, config: MergeConfig):
        self.config = config
        self.ffmpeg_processor = FFmpegProcessor()
        self.synchronizer = AudioVideoSynchronizer()
        self.transition_processor = TransitionProcessor()
        self.quality_controller = QualityController(QualityConfig())
        self.monitor = MergePerformanceMonitor()
    
    async def merge_project(self, 
                          video_project: VideoProject,
                          audio_project: AudioProject,
                          output_path: str) -> MergeResult:
        """合并整个项目"""
        logger.info(f"开始合并项目: {video_project.project_id}")
        
        try:
            # 1. 音视频同步
            with self.monitor.start_operation("synchronization"):
                sync_result = await self.synchronizer.synchronize_segments(
                    video_project.segments,
                    audio_project.segments
                )
            
            # 2. 应用转场效果
            with self.monitor.start_operation("transitions"):
                transition_config = TransitionConfig(
                    default_type=TransitionType(self.config.transition_type),
                    duration=0.5,
                    auto_select=True
                )
                
                processed_segments = await self.transition_processor.apply_transitions(
                    sync_result.adjusted_video_segments,
                    transition_config
                )
            
            # 3. 执行FFmpeg合并
            with self.monitor.start_operation("ffmpeg_merge"):
                merge_result = await self.ffmpeg_processor.merge_segments_with_audio(
                    sync_result.adjusted_video_segments,
                    sync_result.adjusted_audio_segments,
                    output_path,
                    self.config
                )
            
            # 4. 质量验证
            with self.monitor.start_operation("quality_check"):
                quality_report = await self.quality_controller.validate_final_output(output_path)
                merge_result.metadata = {"quality_report": quality_report}
            
            # 5. 生成合并报告
            merge_report = self._generate_merge_report(
                sync_result, processed_segments, merge_result, quality_report
            )
            
            logger.info(f"项目合并完成: {output_path}")
            return merge_result
            
        except Exception as e:
            logger.error(f"项目合并失败: {e}")
            raise MergeError(f"合并失败: {e}")
    
    def _generate_merge_report(self, 
                            sync_result: SyncResult,
                            processed_segments: List[ProcessedSegment],
                            merge_result: MergeResult,
                            quality_report: QualityReport) -> Dict[str, Any]:
        """生成合并报告"""
        return {
            "project_summary": {
                "total_segments": len(processed_segments),
                "total_duration": merge_result.duration,
                "output_file_size": merge_result.file_size,
                "success": merge_result.success
            },
            "sync_summary": sync_result.report,
            "transition_summary": {
                "transition_types_used": list(set(s.transition_type for s in processed_segments)),
                "total_transitions": len(processed_segments) - 1
            },
            "quality_summary": {
                "overall_score": quality_report.overall_score,
                "issues_count": len(quality_report.issues),
                "quality_grade": self._get_quality_grade(quality_report.overall_score)
            },
            "performance_summary": self.monitor.get_performance_summary()
        }
    
    def _get_quality_grade(self, score: float) -> str:
        """获取质量等级"""
        if score >= 0.9:
            return "A+"
        elif score >= 0.8:
            return "A"
        elif score >= 0.7:
            return "B"
        elif score >= 0.6:
            return "C"
        else:
            return "D"

# aggregator/batch_processor.py
import asyncio
from typing import List, Dict, Any, Optional
from dataclasses import dataclass, field
from enum import Enum

class BatchStatus(Enum):
    QUEUED = "queued"
    PROCESSING = "processing"
    COMPLETED = "completed"
    FAILED = "failed"
    CANCELLED = "cancelled"

@dataclass
class BatchJob:
    job_id: str
    video_project: VideoProject
    audio_project: AudioProject
    output_path: str
    config: MergeConfig
    status: BatchStatus = BatchStatus.QUEUED
    created_at: datetime = field(default_factory=datetime.now)
    started_at: Optional[datetime] = None
    completed_at: Optional[datetime] = None
    result: Optional[MergeResult] = None
    error: Optional[str] = None

class BatchProcessor:
    """批处理器"""
    
    def __init__(self, max_concurrent_jobs: int = 2):
        self.max_concurrent_jobs = max_concurrent_jobs
        self.job_queue: asyncio.Queue = asyncio.Queue()
        self.active_jobs: Dict[str, BatchJob] = {}
        self.completed_jobs: Dict[str, BatchJob] = {}
        self.pipeline: Optional[VideoMergePipeline] = None
        self.running = False
    
    async def start(self):
        """启动批处理器"""
        self.running = True
        self.pipeline = VideoMergePipeline(MergeConfig())
        
        # 启动工作协程
        workers = []
        for i in range(self.max_concurrent_jobs):
            worker = asyncio.create_task(self._worker(f"worker-{i}"))
            workers.append(worker)
        
        logger.info(f"批处理器启动，{self.max_concurrent_jobs} 个工作线程")
        
        # 等待所有工作协程完成
        await asyncio.gather(*workers)
    
    async def stop(self):
        """停止批处理器"""
        self.running = False
        
        # 取消所有活动任务
        for job in self.active_jobs.values():
            if job.status == BatchStatus.PROCESSING:
                job.status = BatchStatus.CANCELLED
        
        logger.info("批处理器已停止")
    
    async def submit_job(self, 
                        video_project: VideoProject,
                        audio_project: AudioProject,
                        output_path: str,
                        config: Optional[MergeConfig] = None) -> str:
        """提交批处理任务"""
        job_id = f"job_{datetime.now().strftime('%Y%m%d_%H%M%S')}_{len(self.active_jobs)}"
        
        job = BatchJob(
            job_id=job_id,
            video_project=video_project,
            audio_project=audio_project,
            output_path=output_path,
            config=config or MergeConfig()
        )
        
        await self.job_queue.put(job)
        logger.info(f"任务已提交: {job_id}")
        
        return job_id
    
    async def get_job_status(self, job_id: str) -> Optional[BatchJob]:
        """获取任务状态"""
        if job_id in self.active_jobs:
            return self.active_jobs[job_id]
        elif job_id in self.completed_jobs:
            return self.completed_jobs[job_id]
        else:
            return None
    
    async def cancel_job(self, job_id: str) -> bool:
        """取消任务"""
        if job_id in self.active_jobs:
            job = self.active_jobs[job_id]
            if job.status == BatchStatus.PROCESSING:
                job.status = BatchStatus.CANCELLED
                return True
        return False
    
    async def _worker(self, worker_name: str):
        """工作协程"""
        logger.info(f"工作协程启动: {worker_name}")
        
        while self.running:
            try:
                # 获取任务
                job = await asyncio.wait_for(
                    self.job_queue.get(), 
                    timeout=1.0
                )
                
                # 处理任务
                await self._process_job(job, worker_name)
                
            except asyncio.TimeoutError:
                continue
            except Exception as e:
                logger.error(f"工作协程异常 {worker_name}: {e}")
        
        logger.info(f"工作协程停止: {worker_name}")
    
    async def _process_job(self, job: BatchJob, worker_name: str):
        """处理单个任务"""
        job.status = BatchStatus.PROCESSING
        job.started_at = datetime.now()
        self.active_jobs[job.job_id] = job
        
        try:
            logger.info(f"{worker_name} 开始处理任务: {job.job_id}")
            
            # 执行合并
            result = await self.pipeline.merge_project(
                job.video_project,
                job.audio_project,
                job.output_path
            )
            
            job.result = result
            job.status = BatchStatus.COMPLETED
            job.completed_at = datetime.now()
            
            logger.info(f"任务完成: {job.job_id}")
            
        except Exception as e:
            job.error = str(e)
            job.status = BatchStatus.FAILED
            job.completed_at = datetime.now()
            
            logger.error(f"任务失败: {job.job_id}, 错误: {e}")
        
        finally:
            # 移动到已完成列表
            if job.job_id in self.active_jobs:
                del self.active_jobs[job.job_id]
            self.completed_jobs[job.job_id] = job
    
    def get_queue_status(self) -> Dict[str, Any]:
        """获取队列状态"""
        return {
            "queue_size": self.job_queue.qsize(),
            "active_jobs": len(self.active_jobs),
            "completed_jobs": len(self.completed_jobs),
            "active_job_ids": list(self.active_jobs.keys())
        }
```

# aggregator/monitoring.py
import time
import psutil
from dataclasses import dataclass
from typing import Dict, List

@dataclass
class MergePerformanceMetrics:
    operation: str
    duration: float
    memory_usage_mb: float
    cpu_usage_percent: float
    job_id: str
    success: bool
    error_message: Optional[str] = None

class MergePerformanceMonitor:
    """合并性能监控"""
    
    def __init__(self):
        self.metrics: List[MergePerformanceMetrics] = []
    
    def start_operation(self, operation: str, job_id: str = "") -> 'MergeOperationMonitor':
        """开始监控操作"""
        return MergeOperationMonitor(self, operation, job_id)
    
    def record_metric(self, metric: MergePerformanceMetrics):
        """记录性能指标"""
        self.metrics.append(metric)
    
    def get_performance_summary(self) -> Dict[str, Any]:
        """获取性能摘要"""
        if not self.metrics:
            return {}
        
        # 按操作类型分组
        operations = {}
        for metric in self.metrics:
            if metric.operation not in operations:
                operations[metric.operation] = []
            operations[metric.operation].append(metric)
        
        summary = {}
        for operation, metrics in operations.items():
            durations = [m.duration for m in metrics]
            memory_usage = [m.memory_usage_mb for m in metrics]
            cpu_usage = [m.cpu_usage_percent for m in metrics]
            success_count = sum(1 for m in metrics if m.success)
            
            summary[operation] = {
                "total_operations": len(metrics),
                "success_rate": success_count / len(metrics),
                "avg_duration": np.mean(durations),
                "max_memory": max(memory_usage) if memory_usage else 0,
                "avg_cpu": np.mean(cpu_usage) if cpu_usage else 0
            }
        
        return summary

class MergeOperationMonitor:
    """合并操作监控上下文管理器"""
    
    def __init__(self, monitor: MergePerformanceMonitor, operation: str, job_id: str = ""):
        self.monitor = monitor
        self.operation = operation
        self.job_id = job_id
        self.start_time = None
        self.start_memory = None
    
    def __enter__(self):
        self.start_time = time.time()
        self.start_memory = psutil.Process().memory_info().rss / 1024 / 1024
        return self
    
    def __exit__(self, exc_type, exc_val, exc_tb):
        duration = time.time() - self.start_time
        end_memory = psutil.Process().memory_info().rss / 1024 / 1024
        cpu_usage = psutil.cpu_percent()
        success = exc_type is None
        error_message = str(exc_val) if exc_val else None
        
        metric = MergePerformanceMetrics(
            operation=self.operation,
            duration=duration,
            memory_usage_mb=end_memory,
            cpu_usage_percent=cpu_usage,
            job_id=self.job_id,
            success=success,
            error_message=error_message
        )
        
        self.monitor.record_metric(metric)
