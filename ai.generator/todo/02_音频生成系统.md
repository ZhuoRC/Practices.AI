import asyncio
import aiohttp
from typing import Dict, Any

class AudioAudioIntegration:
    """集成ai.audio系统的TTS服务"""
    
    def __init__(self, audio_service_url: str = "http://localhost:7000"):
        self.base_url = audio_service_url
        self.session = None
    
    async def __aenter__(self):
        self.session = aiohttp.ClientSession()
        return self
    
    async def __aexit__(self, exc_type, exc_val, exc_tb):
        if self.session:
            await self.session.close()
    
    async def get_providers(self) -> Dict[str, Any]:
        """获取可用的TTS提供商"""
        async with self.session.get(f"{self.base_url}/providers") as response:
            return await response.json()
    
    async def get_voices(self, provider: str) -> Dict[str, Any]:
        """获取指定提供商的语音列表"""
        async with self.session.get(f"{self.base_url}/providers/{provider}/voices") as response:
            return await response.json()
    
    async def generate_speech(self, request: TTSRequest) -> AudioResult:
        """生成语音"""
        payload = {
            "text": request.text,
            "provider": request.provider,
            "voice_id": request.voice_id,
            "speed": request.speed,
            "pitch": request.pitch,
            "volume": request.volume,
            "output_format": request.output_format
        }
        
        async with self.session.post(f"{self.base_url}/generate", json=payload) as response:
            data = await response.json()
            return AudioResult(
                audio_id=data["audio_id"],
                duration=data["duration"],
                file_size=data["file_size"],
                format=data["format"],
                provider_used=data["provider_used"],
                voice_used=data["voice_used"],
                download_url=f"{self.base_url}{data['download_url']}"
            )
    
    async def download_audio(self, audio_result: AudioResult, save_path: str) -> str:
        """下载音频文件"""
        async with self.session.get(audio_result.download_url) as response:
            content = await response.read()
            
            with open(save_path, "wb") as f:
                f.write(content)
            
            return save_path

```python
# audio_generator/segment_processor.py
import asyncio
from concurrent.futures import ThreadPoolExecutor
from typing import List, Dict

class SegmentProcessor:
    """段落音频处理器"""
    
    def __init__(self, tts_manager: TTSClientManager, max_concurrent: int = 3):
        self.tts_manager = tts_manager
        self.max_concurrent = max_concurrent
        self.semaphore = asyncio.Semaphore(max_concurrent)
        
    async def process_segments(self, segments: List[Segment], output_dir: str) -> List[ProcessedAudio]:
        """批量处理段落音频生成"""
        tasks = []
        
        for segment in segments:
            task = self._process_single_segment(segment, output_dir)
            tasks.append(task)
        
        # 使用信号量控制并发数
        limited_tasks = [
            self._with_semaphore(task) for task in tasks
        ]
        
        results = await asyncio.gather(*limited_tasks, return_exceptions=True)
        
        # 处理异常结果
        processed_audios = []
        for i, result in enumerate(results):
            if isinstance(result, Exception):
                logger.error(f"段落 {segments[i].id} 处理失败: {result}")
                # 创建占位音频或重试
                processed_audios.append(await self._create_fallback_audio(segments[i], output_dir))
            else:
                processed_audios.append(result)
        
        return processed_audios
    
    async def _with_semaphore(self, coro):
        """使用信号量限制并发"""
        async with self.semaphore:
            return await coro
    
    async def _process_single_segment(self, segment: Segment, output_dir: str) -> ProcessedAudio:
        """处理单个段落"""
        try:
            # 选择最佳TTS客户端
            voice_config = self._select_voice_for_segment(segment)
            
            # 生成音频
            audio_result = await self.tts_manager.synthesize_segment(
                segment.rewritten_text, 
                voice_config
            )
            
            # 下载音频文件
            filename = f"{segment.index:02d}_{segment.id}.mp3"
            file_path = Path(output_dir) / filename
            
            await self._download_audio_file(audio_result, file_path)
            
            # 验证音频文件
            duration = await self._get_actual_duration(file_path)
            
            return ProcessedAudio(
                segment_id=segment.id,
                file_path=str(file_path),
                duration=duration,
                estimated_duration=segment.estimated_duration,
                voice_used=audio_result.voice_used,
                provider_used=audio_result.provider_used,
                file_size=file_path.stat().st_size
            )
            
        except Exception as e:
            logger.error(f"处理段落 {segment.id} 时发生错误: {e}")
            raise

import librosa
import ffmpeg
from pathlib import Path

class DurationAnalyzer:
    """音频时长分析器"""
    
    @staticmethod
    async def get_audio_duration(file_path: str) -> float:
        """获取音频文件的实际时长"""
        try:
            # 方法1: 使用librosa
            duration = librosa.get_duration(path=file_path)
            return duration
        except Exception as e:
            logger.warning(f"librosa分析失败，尝试ffmpeg: {e}")
            
            try:
                # 方法2: 使用ffmpeg
                probe = ffmpeg.probe(file_path)
                duration = float(probe['streams'][0]['duration'])
                return duration
            except Exception as e2:
                logger.error(f"ffmpeg分析也失败: {e2}")
                raise AudioAnalysisError(f"无法分析音频时长: {file_path}")
    
    @staticmethod
    def estimate_speech_duration(text: str, language: str = "zh") -> float:
        """估算文本的语音时长"""
        if language == "zh":
            # 中文：平均3字/秒
            char_count = len(text.replace(" ", ""))
            return char_count / 3.0
        else:
            # 英文：平均2.5词/秒
            word_count = len(text.split())
            return word_count / 2.5
    
    @staticmethod
    def analyze_timing_accuracy(estimated: float, actual: float) -> Dict[str, float]:
        """分析时长估算的准确性"""
        difference = abs(estimated - actual)
        accuracy = max(0, 1 - (difference / estimated)) if estimated > 0 else 0
        
        return {
            "estimated": estimated,
            "actual": actual,
            "difference": difference,
            "accuracy": accuracy
        }
```

```python
# audio_generator/quality_checker.py
import numpy as np
from typing import Dict, List

class AudioQualityChecker:
    """音频质量检查器"""
    
    def __init__(self, sample_rate: int = 22050):
        self.sample_rate = sample_rate
    
    async def check_audio_quality(self, file_path: str) -> QualityReport:
        """检查音频质量"""
        try:
            # 加载音频
            y, sr = librosa.load(file_path, sr=self.sample_rate)
            
            # 分析各项指标
            quality_metrics = {
                "duration": len(y) / sr,
                "sample_rate": sr,
                "bit_depth": self._estimate_bit_depth(y),
                "snr": self._calculate_snr(y),
                "silence_ratio": self._calculate_silence_ratio(y),
                "peak_amplitude": np.max(np.abs(y)),
                "rms_energy": np.sqrt(np.mean(y**2)),
                "zero_crossing_rate": self._calculate_zcr(y),
                "spectral_centroid": np.mean(librosa.feature.spectral_centroid(y=y, sr=sr))
            }
            
            # 生成质量报告
            return QualityReport(
                file_path=file_path,
                metrics=quality_metrics,
                overall_score=self._calculate_overall_score(quality_metrics),
                issues=self._detect_issues(quality_metrics),
                recommendations=self._generate_recommendations(quality_metrics)
            )
            
        except Exception as e:
            raise AudioQualityError(f"音频质量检查失败: {e}")
    
    def _calculate_snr(self, audio: np.ndarray) -> float:
        """计算信噪比"""
        signal_power = np.mean(audio ** 2)
        noise_power = np.var(audio[np.abs(audio) < np.std(audio) * 0.1])
        return 10 * np.log10(signal_power / noise_power) if noise_power > 0 else float('inf')
    
    def _calculate_silence_ratio(self, audio: np.ndarray, threshold: float = 0.01) -> float:
        """计算静音比例"""
        silence_frames = np.sum(np.abs(audio) < threshold)
        total_frames = len(audio)
        return silence_frames / total_frames
    
    def _detect_issues(self, metrics: Dict[str, float]) -> List[str]:
        """检测音频问题"""
        issues = []
        
        if metrics["snr"] < 20:
            issues.append("信噪比较低，可能有噪音")
        
        if metrics["silence_ratio"] > 0.3:
            issues.append("静音比例过高")
        
        if metrics["peak_amplitude"] > 0.95:
            issues.append("音频可能存在削波")
        
        if metrics["rms_energy"] < 0.1:
            issues.append("音量过低")
        
        return issues
```

class AudioFileManager:
    """音频文件管理器"""
    
    def __init__(self, base_dir: str):
        self.base_dir = Path(base_dir)
        self.base_dir.mkdir(parents=True, exist_ok=True)
        
        # 创建子目录
        (self.base_dir / "segments").mkdir(exist_ok=True)

        (self.base_dir / "cache").mkdir(exist_ok=True)
    
    def get_segment_path(self, segment_id: str, format: str = "mp3") -> Path:
        """获取段落音频文件路径"""
        return self.base_dir / "segments" / f"{segment_id}.{format}"
    
    def organize_files(self, project_id: str) -> Dict[str, List[str]]:
        """整理项目文件"""
        project_dir = self.base_dir / project_id
        project_dir.mkdir(exist_ok=True)
        
        # 按类型整理文件
        organized = {
            "audio_segments": [],
            "metadata": [],
            "cache": []
        }
        
        for file_path in self.base_dir.glob(f"{project_id}_*"):
            if file_path.suffix in [".mp3", ".wav"]:
                target = project_dir / "audio_segments" / file_path.name
            elif file_path.suffix in [".json", ".yaml"]:
                target = project_dir / "metadata" / file_path.name
            else:
                target = project_dir / "cache" / file_path.name
            
            target.parent.mkdir(exist_ok=True)
            shutil.move(str(file_path), str(target))
            organized[str(target.parent.name)].append(str(target))
        
        return organized
    
    def cleanup_cache(self, max_age_hours: int = 24):
        """清理缓存文件"""
        cache_dir = self.base_dir / "cache"
        current_time = datetime.now()
        
        for file_path in cache_dir.glob("*"):
            if file_path.is_file():
                file_age = current_time - datetime.fromtimestamp(file_path.stat().st_mtime)
                if file_age.total_seconds() > max_age_hours * 3600:
                    file_path.unlink()
                    logger.info(f"清理缓存文件: {file_path}")

# audio_generator/pipeline.py
class AudioGenerationPipeline:
    """音频生成管道"""
    
    def __init__(self, config: AudioConfig):
        self.config = config
        self.tts_manager = TTSClientManager()
        self.segment_processor = SegmentProcessor(self.tts_manager)
        self.file_manager = AudioFileManager(config.output_directory)
        self.quality_checker = AudioQualityChecker()
    
    async def generate_project_audio(self, script: ProcessedScript) -> AudioProject:
        """为整个项目生成音频"""
        logger.info(f"开始为项目 {script.metadata.title} 生成音频")
        
        # 初始化TTS客户端
        await self.tts_manager.initialize_clients(self.config.tts_config)
        
        # 处理所有段落
        processed_audios = await self.segment_processor.process_segments(
            script.segments,
            self.file_manager.base_dir / "segments"
        )
        
        # 质量检查
        for audio in processed_audios:
            try:
                quality_report = await self.quality_checker.check_audio_quality(audio.file_path)
                audio.quality_score = quality_report.overall_score
                
                if quality_report.issues:
                    logger.warning(f"音频 {audio.segment_id} 存在质量问题: {quality_report.issues}")
            except Exception as e:
                logger.error(f"音频质量检查失败 {audio.segment_id}: {e}")
        
        # 创建音频项目
        audio_project = AudioProject(
            project_id=script.metadata.title,
            script_id=script.metadata.title,
            segments=self._create_audio_segments(processed_audios, script.segments),
            total_duration=sum(audio.duration for audio in processed_audios),
            output_directory=str(self.file_manager.base_dir),
            created_at=datetime.now(),
            metadata={
                "total_segments": len(processed_audios),
                "average_quality": np.mean([audio.quality_score for audio in processed_audios if audio.quality_score]),
                "tts_providers_used": list(set(audio.provider_used for audio in processed_audios))
            }
        )
        
        # 保存项目元数据
        await self._save_project_metadata(audio_project)
        
        logger.info(f"音频生成完成，总时长: {audio_project.total_duration:.2f}秒")
        return audio_project
    
    def _create_audio_segments(self, processed_audios: List[ProcessedAudio], script_segments: List[Segment]) -> List[AudioSegment]:
        """创建音频段落对象"""
        script_segment_map = {seg.id: seg for seg in script_segments}
        
        audio_segments = []
        for audio in processed_audios:
            script_segment = script_segment_map.get(audio.segment_id)
            if script_segment:
                audio_segment = AudioSegment(
                    id=audio.segment_id,
                    segment_index=script_segment.index,
                    file_path=audio.file_path,
                    duration=audio.duration,
                    text=script_segment.rewritten_text,
                    voice_config=self._extract_voice_config(audio)
                )
                audio_segments.append(audio_segment)
        
        return sorted(audio_segments, key=lambda x: x.segment_index)

# audio_generator/retry.py
import asyncio
import functools
from typing import Callable, Any

def async_retry(max_attempts: int = 3, delay: float = 1.0, backoff: float = 2.0):
    """异步重试装饰器"""
    def decorator(func: Callable) -> Callable:
        @functools.wraps(func)
        async def wrapper(*args, **kwargs) -> Any:
            last_exception = None
            current_delay = delay
            
            for attempt in range(max_attempts):
                try:
                    return await func(*args, **kwargs)
                except Exception as e:
                    last_exception = e
                    
                    if attempt == max_attempts - 1:
                        break
                    
                    logger.warning(f"第 {attempt + 1} 次尝试失败，{current_delay}秒后重试: {e}")
                    await asyncio.sleep(current_delay)
                    current_delay *= backoff
            
            raise last_exception
        
        return wrapper
    return decorator

# 使用示例
@async_retry(max_attempts=3, delay=1.0, backoff=2.0)
async def synthesize_with_retry(self, request: TTSRequest) -> AudioResult:
    """带重试的语音合成"""
    return await self.tts_client.synthesize(request)

# audio_generator/monitoring.py
import time
from dataclasses import dataclass
from typing import Dict, List

@dataclass
class PerformanceMetrics:
    operation: str
    duration: float
    success: bool
    error_message: Optional[str] = None
    metadata: Optional[Dict[str, Any]] = None

class PerformanceMonitor:
    """性能监控器"""
    
    def __init__(self):
        self.metrics: List[PerformanceMetrics] = []
    
    def record_operation(self, operation: str, duration: float, success: bool, 
                        error_message: Optional[str] = None, 
                        metadata: Optional[Dict[str, Any]] = None):
        """记录操作性能"""
        metric = PerformanceMetrics(
            operation=operation,
            duration=duration,
            success=success,
            error_message=error_message,
            metadata=metadata
        )
        self.metrics.append(metric)
    
    def get_performance_summary(self) -> Dict[str, Any]:
        """获取性能摘要"""
        if not self.metrics:
            return {}
        
        operations = {}
        for metric in self.metrics:
            if metric.operation not in operations:
                operations[metric.operation] = []
            operations[metric.operation].append(metric)
        
        summary = {}
        for operation, metrics in operations.items():
            durations = [m.duration for m in metrics]
            success_count = sum(1 for m in metrics if m.success)
            
            summary[operation] = {
                "total_operations": len(metrics),
                "success_rate": success_count / len(metrics),
                "avg_duration": np.mean(durations),
                "min_duration": min(durations),
                "max_duration": max(durations),
                "total_duration": sum(durations)
            }
        
        return summary

# 上下文管理器用于监控
class MonitoredOperation:
    """监控操作上下文管理器"""
    
    def __init__(self, monitor: PerformanceMonitor, operation: str, 
                 metadata: Optional[Dict[str, Any]] = None):
        self.monitor = monitor
        self.operation = operation
        self.metadata = metadata
        self.start_time = None
    
    def __enter__(self):
        self.start_time = time.time()
        return self
    
    def __exit__(self, exc_type, exc_val, exc_tb):
        duration = time.time() - self.start_time
        success = exc_type is None
        error_message = str(exc_val) if exc_val else None
        
        self.monitor.record_operation(
            self.operation, duration, success, error_message, self.metadata
        )
