# 阶段3：视频生成系统分析和实现计划

## 1. 需求分析

### 核心目标
- 基于音频和脚本生成纯文字视频
- 实现关键词高亮和动态效果
- 提供简单的背景动画
- 生成大字字幕，确保可读性
- 支持多种视觉风格和主题

### 输入输出
**输入：**
- 结构化脚本（来自阶段1）
- 音频文件和时长信息（来自阶段2）
- 视频样式配置（字体、颜色、背景等）

**输出：**
- 分段视频文件（MP4格式）
- 视频元数据（分辨率、帧率、时长等）
- 字幕文件（SRT/VTT格式）

```python
class VideoGenerator:
    """视频生成引擎核心类"""
    
    def __init__(self, config: VideoConfig):
        self.config = config
        self.renderer = TextRenderer(config)
        self.animator = BackgroundAnimator(config)
        self.subtitle_engine = SubtitleEngine(config)
        
    async def generate_segment_video(self, segment: Segment, audio_path: str) -> VideoSegment:
        """生成单个段落的视频"""
        # 1. 分析音频时长
        duration = await self._get_audio_duration(audio_path)
        
        # 2. 生成文字内容
        text_content = self._prepare_text_content(segment)
        
        # 3. 创建背景动画
        background_clip = await self.animator.create_background(duration, segment.emotion)
        
        # 4. 渲染文字和关键词
        text_clip = await self.renderer.render_text(text_content, duration)
        
        # 5. 合成最终视频
        final_clip = self._compose_video(background_clip, text_clip, duration)
        
        # 6. 导出视频文件
        output_path = self._get_output_path(segment)
        await self._export_video(final_clip, output_path)
        
        return VideoSegment(
            segment_id=segment.id,
            file_path=output_path,
            duration=duration,
            resolution=self.config.resolution,
            fps=self.config.fps
        )
    
    async def generate_subtitle_file(self, segments: List[Segment]) -> str:
        """生成字幕文件"""
        return await self.subtitle_engine.generate_srt(segments)
```

### 2.2 文字渲染器

#### 动态文字效果
```python
# video_generator/text_renderer.py
from moviepy.editor import TextClip, CompositeVideoClip, ColorClip
from typing import List, Tuple
import numpy as np

class TextRenderer:
    """文字渲染器"""
    
    def __init__(self, config: VideoConfig):
        self.config = config
        self.font_manager = FontManager()
        
    async def render_text(self, text_content: TextContent, duration: float) -> CompositeVideoClip:
        """渲染文字内容"""
        clips = []
        
        # 1. 渲染主文字
        main_text_clip = await self._render_main_text(text_content.main_text, duration)
        clips.append(main_text_clip)
        
        # 2. 渲染关键词高亮
        keyword_clips = await self._render_keywords(text_content.keywords, duration)
        clips.extend(keyword_clips)
        
        # 3. 渲染段落数字/标题
        if text_content.title:
            title_clip = await self._render_title(text_content.title, duration)
            clips.append(title_clip)
        
        # 4. 合成所有文字层
        return CompositeVideoClip(clips, size=self.config.resolution)
    
    async def _render_main_text(self, text: str, duration: float) -> TextClip:
        """渲染主文字"""
        # 文字分段处理，避免单行过长
        lines = self._split_text_lines(text, max_chars_per_line=20)
        
        return TextClip(
            text="\n".join(lines),
            fontsize=self.config.font_size_large,
            font=self.font_manager.get_font("chinese_bold"),
            color=self.config.text_color,
            bg_color=None,  # 透明背景
            size=self.config.resolution,
            align='center',
            method='caption'
        ).set_duration(duration).set_position('center')
    
    async def _render_keywords(self, keywords: List[KeywordInfo], duration: float) -> List[TextClip]:
        """渲染关键词高亮"""
        keyword_clips = []
        
        for keyword in keywords:
            # 创建关键词高亮效果
            highlight_clip = TextClip(
                text=keyword.word,
                fontsize=self.config.font_size_keyword,
                font=self.font_manager.get_font("chinese_highlight"),
                color=self.config.keyword_color,
                bg_color=self.config.keyword_bg_color,
                method='caption'
            )
            
            # 添加动画效果
            animated_clip = self._add_keyword_animation(highlight_clip, keyword, duration)
            keyword_clips.append(animated_clip)
        
        return keyword_clips
    
    def _add_keyword_animation(self, clip: TextClip, keyword: KeywordInfo, duration: float) -> TextClip:
        """为关键词添加动画效果"""
        # 1. 缩放动画
        scale_clip = clip.resize(lambda t: 1 + 0.1 * np.sin(2 * np.pi * t / 2))
        
        # 2. 闪烁效果
        def glow_effect(t):
            if t < keyword.start_time or t > keyword.end_time:
                return 1.0
            progress = (t - keyword.start_time) / (keyword.end_time - keyword.start_time)
            return 1.0 + 0.3 * np.sin(np.pi * progress)
        
        glow_clip = scale_clip.set_opacity(glow_effect)
        
        return glow_clip.set_duration(duration)
    
    def _split_text_lines(self, text: str, max_chars_per_line: int) -> List[str]:
        """智能分行"""
        if len(text) <= max_chars_per_line:
            return [text]
        
        lines = []
        current_line = ""
        
        for char in text:
            if len(current_line) >= max_chars_per_line:
                # 尝试在标点符号处分行
                punctuation_positions = [i for i, c in enumerate(current_line) if c in '，。！？；：']
                if punctuation_positions:
                    split_pos = max(punctuation_positions) + 1
                    lines.append(current_line[:split_pos])
                    current_line = current_line[split_pos:] + char
                else:
                    lines.append(current_line)
                    current_line = char
            else:
                current_line += char
        
        if current_line:
            lines.append(current_line)
        
        return lines
```

### 2.3 背景动画器

#### 动态背景生成
```python
# video_generator/background_animator.py
from moviepy.editor import ColorClip, VideoClip
import numpy as np
from typing import Optional

class BackgroundAnimator:
    """背景动画器"""
    
    def __init__(self, config: VideoConfig):
        self.config = config
        self.animation_presets = self._load_animation_presets()
    
    async def create_background(self, duration: float, emotion: str = "neutral") -> VideoClip:
        """创建背景动画"""
        preset = self._get_emotion_preset(emotion)
        
        if preset["type"] == "gradient":
            return await self._create_gradient_background(duration, preset)
        elif preset["type"] == "particles":
            return await self._create_particle_background(duration, preset)
        elif preset["type"] == "waves":
            return await self._create_wave_background(duration, preset)
        else:
            return await self._create_static_background(duration, preset)
    
    async def _create_gradient_background(self, duration: float, preset: dict) -> VideoClip:
        """创建渐变背景"""
        def make_frame(t):
            # 时间相关的颜色变化
            progress = t / duration
            
            # 从颜色A到颜色B的渐变
            color_a = np.array(preset["color_a"])
            color_b = np.array(preset["color_b"])
            
            # 添加时间变化
            color_shift = 0.1 * np.sin(2 * np.pi * progress * preset["speed"])
            current_color = color_a * (1 - progress) + color_b * progress + color_shift
            
            return np.full(
                (self.config.resolution[1], self.config.resolution[0], 3),
                current_color,
                dtype=np.uint8
            )
        
        return VideoClip(make_frame, duration=duration)
    
    async def _create_particle_background(self, duration: float, preset: dict) -> VideoClip:
        """创建粒子背景"""
        particles = self._generate_particles(preset["particle_count"])
        
        def make_frame(t):
            # 创建基础背景
            frame = np.full(
                (self.config.resolution[1], self.config.resolution[0], 3),
                preset["base_color"],
                dtype=np.uint8
            )
            
            # 绘制粒子
            for particle in particles:
                # 更新粒子位置
                particle["x"] += particle["vx"] * t
                particle["y"] += particle["vy"] * t
                
                # 边界检查
                if particle["x"] < 0 or particle["x"] > self.config.resolution[0]:
                    particle["vx"] *= -1
                if particle["y"] < 0 or particle["y"] > self.config.resolution[1]:
                    particle["vy"] *= -1
                
                # 绘制粒子
                x, y = int(particle["x"]), int(particle["y"])
                if 0 <= x < self.config.resolution[0] and 0 <= y < self.config.resolution[1]:
                    radius = int(particle["size"])
                    for i in range(max(0, y-radius), min(self.config.resolution[1], y+radius+1)):
                        for j in range(max(0, x-radius), min(self.config.resolution[0], x+radius+1)):
                            if (i-y)**2 + (j-x)**2 <= radius**2:
                                frame[i, j] = particle["color"]
            
            return frame
        
        return VideoClip(make_frame, duration=duration)
    
    async def _create_wave_background(self, duration: float, preset: dict) -> VideoClip:
        """创建波浪背景"""
        def make_frame(t):
            # 创建网格
            x = np.linspace(0, 2*np.pi, self.config.resolution[0])
            y = np.linspace(0, 2*np.pi, self.config.resolution[1])
            X, Y = np.meshgrid(x, y)
            
            # 时间相关的波浪
            Z = np.sin(X + preset["frequency_x"] * t) * np.cos(Y + preset["frequency_y"] * t)
            
            # 归一化到0-255
            Z_normalized = ((Z + 1) * 127.5).astype(np.uint8)
            
            # 应用颜色映射
            frame = np.zeros((self.config.resolution[1], self.config.resolution[0], 3), dtype=np.uint8)
            frame[:, :, 0] = Z_normalized * preset["color_r"]
            frame[:, :, 1] = Z_normalized * preset["color_g"]
            frame[:, :, 2] = Z_normalized * preset["color_b"]
            
            return frame
        
        return VideoClip(make_frame, duration=duration)
    
    def _generate_particles(self, count: int) -> List[dict]:
        """生成粒子"""
        particles = []
        for _ in range(count):
            particle = {
                "x": np.random.uniform(0, self.config.resolution[0]),
                "y": np.random.uniform(0, self.config.resolution[1]),
                "vx": np.random.uniform(-50, 50),  # 像素/秒
                "vy": np.random.uniform(-50, 50),
                "size": np.random.uniform(1, 5),
                "color": np.random.randint(100, 255, 3)
            }
            particles.append(particle)
        return particles
    
    def _get_emotion_preset(self, emotion: str) -> dict:
        """根据情感获取预设"""
        presets = {
            "neutral": {
                "type": "gradient",
                "color_a": [30, 30, 40],
                "color_b": [50, 50, 60],
                "speed": 0.5
            },
            "excited": {
                "type": "particles",
                "particle_count": 50,
                "base_color": [20, 20, 30],
                "speed": 2.0
            },
            "serious": {
                "type": "gradient",
                "color_a": [20, 20, 30],
                "color_b": [40, 40, 50],
                "speed": 0.2
            },
            "calm": {
                "type": "waves",
                "frequency_x": 1.0,
                "frequency_y": 0.5,
                "color_r": 0.3,
                "color_g": 0.5,
                "color_b": 0.7
            }
        }
        return presets.get(emotion, presets["neutral"])
```

```python
# video_generator/subtitle_engine.py
from datetime import timedelta
from typing import List, Dict

class SubtitleEngine:
    """字幕引擎"""
    
    def __init__(self, config: VideoConfig):
        self.config = config
    
    async def generate_srt(self, segments: List[Segment]) -> str:
        """生成SRT格式字幕"""
        srt_content = []
        current_time = 0.0
        
        for i, segment in enumerate(segments):
            # 计算时间范围
            start_time = current_time
            end_time = current_time + segment.estimated_duration
            
            # 转换为SRT时间格式
            start_srt = self._seconds_to_srt_time(start_time)
            end_srt = self._seconds_to_srt_time(end_time)
            
            # 添加字幕条目
            srt_entry = f"""{i + 1}
{start_srt} --> {end_srt}
{segment.rewritten_text}

"""
            srt_content.append(srt_entry)
            
            current_time = end_time
        
        return "".join(srt_content)
    
    async def generate_vtt(self, segments: List[Segment]) -> str:
        """生成VTT格式字幕"""
        vtt_content = ["WEBVTT\n\n"]
        current_time = 0.0
        
        for segment in segments:
            start_time = current_time
            end_time = current_time + segment.estimated_duration
            
            start_vtt = self._seconds_to_vtt_time(start_time)
            end_vtt = self._seconds_to_vtt_time(end_time)
            
            vtt_entry = f"{start_vtt} --> {end_vtt}\n{segment.rewritten_text}\n\n"
            vtt_content.append(vtt_entry)
            
            current_time = end_time
        
        return "".join(vtt_content)
    
    def _seconds_to_srt_time(self, seconds: float) -> str:
        """转换为SRT时间格式"""
        td = timedelta(seconds=seconds)
        hours, remainder = divmod(td.total_seconds(), 3600)
        minutes, seconds = divmod(remainder, 60)
        milliseconds = int((seconds % 1) * 1000)
        
        return f"{int(hours):02d}:{int(minutes):02d}:{int(seconds):02d},{milliseconds:03d}"
    
    def _seconds_to_vtt_time(self, seconds: float) -> str:
        """转换为VTT时间格式"""
        td = timedelta(seconds=seconds)
        hours, remainder = divmod(td.total_seconds(), 3600)
        minutes, seconds = divmod(remainder, 60)
        milliseconds = int((seconds % 1) * 1000)
        
        return f"{int(hours):02d}:{int(minutes):02d}:{int(seconds):02d}.{milliseconds:03d}"
    
    async def generate_burned_subtitles(self, segments: List[Segment], video_resolution: Tuple[int, int]) -> List[dict]:
        """生成烧录字幕数据"""
        subtitle_data = []
        current_time = 0.0
        
        for segment in segments:
            start_time = current_time
            end_time = current_time + segment.estimated_duration
            
            subtitle_entry = {
                "text": segment.rewritten_text,
                "start_time": start_time,
                "end_time": end_time,
                "position": "bottom",
                "style": {
                    "font_size": self.config.subtitle_font_size,
                    "color": self.config.subtitle_color,
                    "background_color": self.config.subtitle_bg_color,
                    "font": self.config.subtitle_font
                }
            }
            
            subtitle_data.append(subtitle_entry)
            current_time = end_time
        
        return subtitle_data
```

```python
# video_generator/font_manager.py
from pathlib import Path
from typing import Dict, Optional
import platform

class FontManager:
    """字体管理器"""
    
    def __init__(self):
        self.font_cache = {}
        self.system_fonts = self._scan_system_fonts()
        self.custom_fonts = self._load_custom_fonts()
    
    def get_font(self, font_type: str) -> str:
        """获取字体路径"""
        if font_type in self.font_cache:
            return self.font_cache[font_type]
        
        font_path = self._find_best_font(font_type)
        self.font_cache[font_type] = font_path
        return font_path
    
    def _find_best_font(self, font_type: str) -> str:
        """找到最佳字体"""
        font_preferences = {
            "chinese_bold": [
                "NotoSansCJK-Bold.ttc",
                "SimHei.ttf",
                "Microsoft YaHei Bold",
                "Source Han Sans Bold"
            ],
            "chinese_regular": [
                "NotoSansCJK-Regular.ttc",
                "SimSun.ttf",
                "Microsoft YaHei",
                "Source Han Sans"
            ],
            "chinese_highlight": [
                "NotoSansCJK-Black.ttc",
                "SimHei.ttf",
                "Microsoft YaHei Bold",
                "Source Han Sans Heavy"
            ],
            "english_bold": [
                "Arial Bold.ttf",
                "Helvetica Bold.ttf",
                "Roboto-Bold.ttf",
                "OpenSans-Bold.ttf"
            ]
        }
        
        preferences = font_preferences.get(font_type, ["Arial.ttf"])
        
        # 首先检查自定义字体
        for font_name in preferences:
            if font_name in self.custom_fonts:
                return self.custom_fonts[font_name]
        
        # 然后检查系统字体
        for font_name in preferences:
            if font_name in self.system_fonts:
                return self.system_fonts[font_name]
        
        # 最后返回默认字体
        return "Arial.ttf"
    
    def _scan_system_fonts(self) -> Dict[str, str]:
        """扫描系统字体"""
        system = platform.system()
        font_dirs = []
        
        if system == "Windows":
            font_dirs = ["C:/Windows/Fonts"]
        elif system == "Darwin":  # macOS
            font_dirs = [
                "/System/Library/Fonts",
                "/Library/Fonts",
                "~/Library/Fonts"
            ]
        elif system == "Linux":
            font_dirs = [
                "/usr/share/fonts",
                "/usr/local/share/fonts",
                "~/.fonts"
            ]
        
        fonts = {}
        for font_dir in font_dirs:
            font_path = Path(font_dir).expanduser()
            if font_path.exists():
                for font_file in font_path.rglob("*.ttf"):
                    fonts[font_file.name] = str(font_file)
                for font_file in font_path.rglob("*.ttc"):
                    fonts[font_file.name] = str(font_file)
        
        return fonts
    
    def _load_custom_fonts(self) -> Dict[str, str]:
        """加载自定义字体"""
        custom_fonts = {}
        font_dir = Path(__file__).parent / "assets" / "fonts"
        
        if font_dir.exists():
            for font_file in font_dir.rglob("*.ttf"):
                custom_fonts[font_file.name] = str(font_file)
            for font_file in font_dir.rglob("*.ttc"):
                custom_fonts[font_file.name] = str(font_file)
        
        return custom_fonts
```

```python
# video_generator/pipeline.py
class VideoGenerationPipeline:
    """视频生成管道"""
    
    def __init__(self, config: VideoConfig):
        self.config = config
        self.video_generator = VideoGenerator(config)
        self.subtitle_engine = SubtitleEngine(config)
        self.file_manager = VideoFileManager(config.output_directory)
    
    async def generate_project_videos(self, script: ProcessedScript, audio_project: AudioProject) -> VideoProject:
        """为整个项目生成视频"""
        logger.info(f"开始为项目 {script.metadata.title} 生成视频")
        
        # 创建输出目录
        self.file_manager.prepare_output_directory(script.metadata.title)
        
        # 生成每个段落的视频
        video_segments = []
        for segment in script.segments:
            # 找到对应的音频文件
            audio_segment = self._find_audio_segment(segment.id, audio_project)
            if not audio_segment:
                logger.error(f"找不到段落 {segment.id} 的音频文件")
                continue
            
            # 生成视频
            with MonitoredOperation(self.monitor, "generate_segment_video", {"segment_id": segment.id}):
                video_segment = await self.video_generator.generate_segment_video(
                    segment, audio_segment.file_path
                )
                video_segments.append(video_segment)
        
        # 生成字幕文件
        subtitle_file = await self.subtitle_engine.generate_srt(script.segments)
        subtitle_path = self.file_manager.save_subtitle_file(script.metadata.title, subtitle_file)
        
        # 创建视频项目
        video_project = VideoProject(
            project_id=script.metadata.title,
            segments=video_segments,
            total_duration=sum(segment.duration for segment in video_segments),
            resolution=self.config.resolution,
            fps=self.config.fps,
            output_directory=self.file_manager.get_project_directory(script.metadata.title),
            subtitle_file=subtitle_path,
            metadata={
                "total_segments": len(video_segments),
                "average_segment_duration": np.mean([seg.duration for seg in video_segments]),
                "video_format": self.config.format.value
            }
        )
        
        # 保存项目元数据
        await self._save_project_metadata(video_project)
        
        logger.info(f"视频生成完成，总时长: {video_project.total_duration:.2f}秒")
```

```python
# video_generator/optimizer.py
import asyncio
from concurrent.futures import ThreadPoolExecutor
from functools import lru_cache

class VideoOptimizer:
    """视频生成优化器"""
    
    def __init__(self, max_workers: int = 4):
        self.max_workers = max_workers
        self.executor = ThreadPoolExecutor(max_workers=max_workers)
        self.cache = {}
    
    async def generate_videos_parallel(self, segments: List[Segment], 
                                     audio_segments: List[AudioSegment],
                                     video_generator: VideoGenerator) -> List[VideoSegment]:
        """并行生成视频"""
        # 创建任务列表
        tasks = []
        segment_audio_map = {seg.segment_id: seg for seg in audio_segments}
        
        for segment in segments:
            audio_segment = segment_audio_map.get(segment.id)
            if audio_segment:
                task = self._generate_single_video(segment, audio_segment, video_generator)
                tasks.append(task)
        
        # 并行执行
        results = await asyncio.gather(*tasks, return_exceptions=True)
        
        # 处理结果
        video_segments = []
        for i, result in enumerate(results):
            if isinstance(result, Exception):
                logger.error(f"视频生成失败 {segments[i].id}: {result}")
                # 创建占位视频或重试
                continue
            else:
                video_segments.append(result)
        
        return video_segments
    
    async def _generate_single_video(self, segment: Segment, 
                                    audio_segment: AudioSegment,
                                    video_generator: VideoGenerator) -> VideoSegment:
        """生成单个视频（在独立线程中）"""
        loop = asyncio.get_event_loop()
        
        # 在线程池中执行CPU密集型操作
        return await loop.run_in_executor(
            self.executor,
            self._sync_generate_video,
            segment,
            audio_segment,
            video_generator
        )
    
    def _sync_generate_video(self, segment: Segment, 
                           audio_segment: AudioSegment,
                           video_generator: VideoGenerator) -> VideoSegment:
        """同步生成视频（在线程池中执行）"""
        # 这里调用同步的视频生成方法
        return asyncio.run(video_generator.generate_segment_video(
            segment, audio_segment.file_path
        ))
    
    @lru_cache(maxsize=100)
    def get_cached_background(self, duration: float, emotion: str, preset_hash: str) -> VideoClip:
        """缓存背景动画"""
        cache_key = f"{duration}_{emotion}_{preset_hash}"
        if cache_key in self.cache:
            return self.cache[cache_key]
        
        # 生成新的背景
        background = self._generate_background(duration, emotion)
        self.cache[cache_key] = background
        return background
    
    def clear_cache(self):
        """清理缓存"""
        self.cache.clear()
```

```python
# video_generator/quality_checker.py
import cv2
import numpy as np
from pathlib import Path

class VideoQualityChecker:
    """视频质量检查器"""
    
    def __init__(self, config: VideoConfig):
        self.config = config
        self.quality_thresholds = {
            "min_resolution": (1280, 720),
            "min_fps": 24,
            "max_bitrate_variance": 0.1,
            "min_duration": 1.0
        }
    
    async def check_video_quality(self, video_path: str) -> QualityReport:
        """检查视频质量"""
        try:
            # 使用OpenCV分析视频
            cap = cv2.VideoCapture(video_path)
            
            # 获取基本信息
            fps = cap.get(cv2.CAP_PROP_FPS)
            width = int(cap.get(cv2.CAP_PROP_FRAME_WIDTH))
            height = int(cap.get(cv2.CAP_PROP_FRAME_HEIGHT))
            frame_count = int(cap.get(cv2.CAP_PROP_FRAME_COUNT))
            duration = frame_count / fps if fps > 0 else 0
            
            # 分析视频质量
            quality_metrics = {
                "resolution": (width, height),
                "fps": fps,
                "duration": duration,
                "frame_count": frame_count,
                "estimated_bitrate": self._estimate_bitrate(video_path, duration),
                "quality_score": self._calculate_quality_score(width, height, fps)
            }
            
            # 检测问题
            issues = self._detect_quality_issues(quality_metrics)
            
            cap.release()
            
            return QualityReport(
                file_path=video_path,
                metrics=quality_metrics,
                overall_score=quality_metrics["quality_score"],
                issues=issues,
                recommendations=self._generate_recommendations(issues)
            )
            
        except Exception as e:
            raise VideoQualityError(f"视频质量检查失败: {e}")
    
    def _detect_quality_issues(self, metrics: Dict[str, Any]) -> List[str]:
        """检测质量问题"""
        issues = []
        
        # 分辨率检查
        width, height = metrics["resolution"]
        min_width, min_height = self.quality_thresholds["min_resolution"]
        if width < min_width or height < min_height:
            issues.append(f"分辨率过低 ({width}x{height})")
        
        # 帧率检查
        if metrics["fps"] < self.quality_thresholds["min_fps"]:
            issues.append(f"帧率过低 ({metrics['fps']} FPS)")
        
        # 时长检查
        if metrics["duration"] < self.quality_thresholds["min_duration"]:
            issues.append(f"视频时长过短 ({metrics['duration']:.2f}秒)")
        
        return issues
    
    def _calculate_quality_score(self, width: int, height: int, fps: float) -> float:
        """计算质量分数"""
        # 基于分辨率、帧率等计算质量分数
        resolution_score = min(1.0, (width * height) / (1920 * 1080))
        fps_score = min(1.0, fps / 30)
        
        return (resolution_score * 0.7 + fps_score * 0.3)
```

```python
# video_generator/monitoring.py
import time
import psutil
from dataclasses import dataclass
from typing import Dict, List

@dataclass
class VideoPerformanceMetrics:
    operation: str
    duration: float
    memory_usage_mb: float
    cpu_usage_percent: float
    success: bool
    error_message: Optional[str] = None

class VideoPerformanceMonitor:
    """视频生成性能监控"""
    
    def __init__(self):
        self.metrics: List[VideoPerformanceMetrics] = []
    
    def start_operation(self, operation: str) -> 'OperationMonitor':
        """开始监控操作"""
        return OperationMonitor(self, operation)
    
    def record_metric(self, metric: VideoPerformanceMetrics):
        """记录性能指标"""
        self.metrics.append(metric)
    
    def get_performance_summary(self) -> Dict[str, Any]:
        """获取性能摘要"""
        if not self.metrics:
            return {}
        
        # 按操作类型分组
        operations = {}
        for metric in self.metrics:
            if metric.operation not in operations:
                operations[metric.operation] = []
            operations[metric.operation].append(metric)
        
        summary = {}
        for operation, metrics in operations.items():
            durations = [m.duration for m in metrics]
            memory_usage = [m.memory_usage_mb for m in metrics]
            cpu_usage = [m.cpu_usage_percent for m in metrics]
            success_count = sum(1 for m in metrics if m.success)
            
            summary[operation] = {
                "total_operations": len(metrics),
                "success_rate": success_count / len(metrics),
                "avg_duration": np.mean(durations),
                "max_memory": max(memory_usage),
                "avg_cpu": np.mean(cpu_usage)
            }
        
        return summary

class OperationMonitor:
    """操作监控上下文管理器"""
    
    def __init__(self, monitor: VideoPerformanceMonitor, operation: str):
        self.monitor = monitor
        self.operation = operation
        self.start_time = None
        self.start_memory = None
    
    def __enter__(self):
        self.start_time = time.time()
        self.start_memory = psutil.Process().memory_info().rss / 1024 / 1024
        return self
    
    def __exit__(self, exc_type, exc_val, exc_tb):
        duration = time.time() - self.start_time
        end_memory = psutil.Process().memory_info().rss / 1024 / 1024
        cpu_usage = psutil.cpu_percent()
        success = exc_type is None
        error_message = str(exc_val) if exc_val else None
        
        metric = VideoPerformanceMetrics(
            operation=self.operation,
            duration=duration,
            memory_usage_mb=end_memory,
            cpu_usage_percent=cpu_usage,
            success=success,
            error_message=error_message
        )
        
        self.monitor.record_metric(metric)
```
